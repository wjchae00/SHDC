AddAnimationItem

애니메이션 추가

AddAnimationItem(ids, startTime, duration, startColor, endColor)

Example

// 애니메이션 시작일 지정 vizcore.Animation.SetStartDate(new Date(2022,7,9), 1); //노드 이름에 해당하는 노드 반환 let nodes = vizcore.Object3D.Find.GetNodeByName( 'PIPE101' // Keyword , true // Full Match ); //지정된 노드 하위 BodyID 목록 반환 let ids = vizcore.Object3D.GetBodyIdsByNode(nodes); // 애니메이션 추가 vizcore.Animation.AddAnimationItem( ids // Body IDs(Array or Number) ,0 // 시작 시간 ,10 // 재생 시간 ,new VIZCore.Color(255,0,0,255) //시작 색상 (R, G, B, A) ,new VIZCore.Color(0,255,0,255) //종료 색상 (R, G, B, A) ); // 애니메이션 시작 vizcore.Animation.Start();
